实现

var pieData = [
    {
        name: '淘宝',
        value: 11231
    },
    {
        name: '京东',
        value: 22673
    },
    {
        name: '唯品会',
        value: 6123
    },
    {
        name: '1号店',
        value: 8989
    },
    {
        name: '聚美优品',
        value: 6700
    }
]
var option = {
    series: [
        {
            type: 'pie',
            data: pieData
        }
    ]
}
